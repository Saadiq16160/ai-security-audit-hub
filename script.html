function renderTools() {

  let tools = TOOLS.filter(t => {

    const matchSearch = !currentFilter.search ||

      t.name.toLowerCase().includes(currentFilter.search) ||

      t.desc.toLowerCase().includes(currentFilter.search) ||

      t.domain.toLowerCase().includes(currentFilter.search) ||

      (t.bestFor||'').toLowerCase().includes(currentFilter.search);

    const matchType = currentFilter.type==='all' || t.type===currentFilter.type;

    const matchDomain = currentFilter.domain==='all' || t.domain===currentFilter.domain;

    return matchSearch && matchType && matchDomain;

  });

 

  const grid = document.getElementById('toolGrid');

  const noRes = document.getElementById('noResults');

 

  if(!tools.length){

    grid.innerHTML='';

    noRes.style.display='block';

    return;

  }

  noRes.style.display='none';

    const sc = t.overallScore;

function getIndustryScore(tool, industry) {

  const base = tool.industryScore || 70;

  const warnings = INDUSTRY_WARNINGS[tool.id] || {};

  const hasWarning = !!warnings[industry];

  const inSupportedList = (tool.industries||[]).includes(industry);

  if(hasWarning && !inSupportedList) return Math.max(20, base-30);

  if(hasWarning) return Math.max(30, base-20);

  if(inSupportedList) return Math.min(98, base+8);

  return base;

}

 

function scoreColor(s) {

}

 

function buildGaugeSVG(score, color) {

  const r=50,cx=60,cy=60;

  const startAngle=-210, totalAngle=240;

  const endAngle=startAngle+(score/100)*totalAngle;

  const toRad=d=>d*Math.PI/180;

  const arc=(s,e)=>{

    const sx=cx+r*Math.cos(toRad(s)),sy=cy+r*Math.sin(toRad(s));

    const ex=cx+r*Math.cos(toRad(e)),ey=cy+r*Math.sin(toRad(e));

    const large=e-s>180?1:0;

    return `M${sx} ${sy} A${r} ${r} 0 ${large} 1 ${ex} ${ey}`;

  };

  return `<svg width="120" height="100" viewBox="0 0 120 100" class="score-ring">

    <path d="${arc(startAngle,startAngle+totalAngle)}" fill="none" stroke="#1E3A5F" stroke-width="10" stroke-linecap="round"/>

    <path d="${arc(startAngle,endAngle)}" fill="none" stroke="${color}" stroke-width="10" stroke-linecap="round"/>

  </svg>`;

}

function triggerDownload(html, filename) {

  URL.revokeObjectURL(url);
